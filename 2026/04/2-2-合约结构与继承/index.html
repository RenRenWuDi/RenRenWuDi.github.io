合约结构与继承
合约定义、继承、接口、库
系列二「Solidity智能合约开发」第2篇
💡 你将学到：合约结构、继承机制、抽象合约、接口、库的使用
⏱ 阅读时间：约15分钟

  1. 合约的基本结构
    一个完整的Solidity合约包含以下部分：
    Inheritance.sol
    // SPDX-License-Identifier: MIT
    pragma solidity ^0.8.0;

// 导入其他合约
import “./OtherContract.sol”;

// 合约定义
contract MyContract {
// 1. 状态变量
uint256 public value;
address public owner;

// 2. 事件
event ValueChanged(uint256 newValue);

// 3. 修饰器
modifier onlyOwner() {
require(msg.sender == owner, “Not owner”);
_; // 执行被修饰的函数
}

// 4. 构造函数
constructor(uint256 _initialValue) {
value = _initialValue;
owner = msg.sender;
}

// 5. 函数
function setValue(uint256 _value) public onlyOwner {
value = _value;
emit ValueChanged(_value);
}
}

各部分说明
SPDX-License-Identifier：开源许可证
pragma：指定编译器版本
import：导入其他文件
状态变量：永久存储在区块链上
事件：记录日志，前端可监听
修饰器：复用代码逻辑
构造函数：部署时执行一次
2. 合约继承
Solidity支持单继承和多继承（通过线性化实现）。
基本继承语法
Inheritance.sol
// 父合约
contract Animal {
string public name;

constructor(string memory _name) {
name = _name;
}

function speak() public virtual returns (string memory) {
return “…”;
}
}

// 子合约继承父合约
contract Dog is Animal {
constructor(string memory _name) Animal(_name) {}

function speak() public pure override returns (string memory) {
return “Woof!”;
}
}

继承要点：

• is 关键字表示继承
• 子合约可以访问父合约的public/internal成员
• virtual：父合约函数可被重写
• override：子合约重写父合约函数
• 构造函数需要显式调用父合约构造函数
多继承
Inheritance.sol
contract A { function foo() public virtual {} }
contract B { function foo() public virtual {} }

// 多继承：按从右到左顺序线性化
contract C is A, B {
function foo() public override(A, B) {}
}

💡 C3线性化：Solidity使用Python的C3算法解决多继承顺序。最基类放在最右边。
调用父合约函数
Inheritance.sol
contract Base {
function foo() public virtual returns (string memory) {
return “Base”;
}
}

contract Derived is Base {
function foo() public override returns (string memory) {
// 调用父合约函数
return string(abi.encodePacked(super.foo(), “ + Derived”));
}
}

  1. 抽象合约与接口
    抽象合约（Abstract Contract）
    包含至少一个未实现函数的合约，必须用abstract标记，不能直接部署。
    Inheritance.sol
    abstract contract Animal {
    string public name;

// 已实现函数
function setName(string memory _name) public {
name = _name;
}

// 未实现函数（虚函数）
function speak() public virtual returns (string memory);
}

contract Cat is Animal {
function speak() public pure override returns (string memory) {
return “Meow!”;
}
}

接口（Interface）
接口特点：

• 所有函数必须是external且没有实现
• 不能定义状态变量
• 不能定义构造函数
• 可以定义事件和枚举
• 用于定义标准（如ERC20、ERC721）
Inheritance.sol
// ERC20标准接口
interface IERC20 {
function totalSupply() external view returns (uint256);
function balanceOf(address account) external view returns (uint256);
function transfer(address to, uint256 amount) external returns (bool);
function allowance(address owner, address spender) external view returns (uint256);
function approve(address spender, uint256 amount) external returns (bool);
function transferFrom(address from, address to, uint256 amount) external returns (bool);

event Transfer(address indexed from, address indexed to, uint256 value);
event Approval(address indexed owner, address indexed spender, uint256 value);
}

接口的使用
Inheritance.sol
// 与任何ERC20代币交互
function getTokenBalance(address token, address user) public view returns (uint256) {
return IERC20(token).balanceOf(user);
}

// 转账任意ERC20代币
function transferToken(address token, address to, uint256 amount) public {
IERC20(token).transfer(to, amount);
}

  1. 库（Library）
    库是特殊的合约，用于封装可复用的函数。
    库的特点：

• 不能定义状态变量
• 不能继承或被继承
• 不能接收ETH
• 部署一次，多次复用
• 使用using for扩展类型
创建库
Inheritance.sol
// SafeMath库（Solidity 0.8前常用，现在内置溢出检查）
library SafeMath {
function add(uint256 a, uint256 b) internal pure returns (uint256) {
uint256 c = a + b;
require(c >= a, “Addition overflow”);
return c;
}

function sub(uint256 a, uint256 b) internal pure returns (uint256) {
require(b <= a, “Subtraction overflow”);
return a - b;
}

function mul(uint256 a, uint256 b) internal pure returns (uint256) {
if (a == 0) return 0;
uint256 c = a * b;
require(c / a == b, “Multiplication overflow”);
return c;
}
}

使用库
Inheritance.sol
// 方式1：直接调用
uint256 result = SafeMath.add(a, b);

// 方式2：using for（更优雅）
using SafeMath for uint256;

uint256 result = a.add(b); // 等价于 SafeMath.add(a, b)
uint256 diff = a.sub(b); // 等价于 SafeMath.sub(a, b)

常用内置库
Strings：字符串处理
Address：地址工具函数
Arrays：数组操作
Create2：确定性地址创建
5. 完整示例：可升级的代币合约
Inheritance.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

// 接口
interface IToken {
function transfer(address to, uint256 amount) external returns (bool);
function balanceOf(address account) external view returns (uint256);
}

// 抽象合约：基础功能
abstract contract Ownable {
address public owner;

constructor() { owner = msg.sender; }

modifier onlyOwner() {
require(msg.sender == owner, “Not owner”);
_;
}
}

// 库
library SafeMath {
function add(uint256 a, uint256 b) internal pure returns (uint256) {
return a + b; // 0.8+内置溢出检查
}
}

// 主合约：继承 + 实现接口
contract MyToken is IToken, Ownable {
using SafeMath for uint256;

string public name = “MyToken”;
string public symbol = “MTK”;
uint8 public decimals = 18;
uint256 public totalSupply;

mapping(address => uint256) public balanceOf;

constructor(uint256 initialSupply) {
totalSupply = initialSupply;
balanceOf[msg.sender] = initialSupply;
}

function transfer(address to, uint256 amount) public returns (bool) {
require(balanceOf[msg.sender] >= amount, “Insufficient balance”);
balanceOf[msg.sender] -= amount;
balanceOf[to] = balanceOf[to].add(amount);
return true;
}

function mint(address to, uint256 amount) public onlyOwner {
totalSupply = totalSupply.add(amount);
balanceOf[to] = balanceOf[to].add(amount);
}
}

  1. 总结

① 合约由状态变量、事件、修饰器、构造函数、函数组成
② 继承使用is关键字，支持多继承（C3线性化）
③ virtual/override控制函数重写
④ 抽象合约包含未实现函数，不能直接部署
⑤ 接口定义标准，所有函数都是external
⑥ 库封装可复用函数，使用using for扩展类型
